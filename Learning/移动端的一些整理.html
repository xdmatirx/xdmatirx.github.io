  3. 区域长按选中

    长按选中需要借助标签的三个事件

    touchstart, touchend, touchmove

    	touchStart() {
      // let that = this
      clearTimeout(this.timer)
      this.timer = 0
      this.timer = setTimeout(() => {
        // 600毫秒后实现我们想要做的事件
        this.show = true
        console.log('按下了')
        this.$emit('overlayShow')
      }, 600)
    },
    touchEnd() {
      clearTimeout(this.timer)
    },
    touchMove() {
      clearTimeout(this.timer)
    },

  4. loading跳转到完成或失败的过渡动画绘制

	makingRecovery() {
      // 添加正在备份的dom和loading的dom
      // process step保证了我们要展示的dom先生成
      this.processing = true
      this.stepOne = true
      //  准备一下即将出来的文字
      this.processText = '正在备份...'
      this.processDetail = '请勿关闭设备以免造成数据丢失'
      let that = this
      setTimeout(() => {
        document.getElementById('stepOne').className = 'stepOne header toShow'
        document.title = '备份中...'
        let gmId = this.$route.params.gmId
        let message = this.message
        makingBackup(gmId, message)
          .then(res => {
            console.log(res)
            let status = res.data.status
            document.getElementById('stepOne').className = 'stepOne header'
            setTimeout(() => {
              this.stepOne = false
            }, 1000)
            switch (status) {
              case 0:
                that.successResult()
                break
              case -4:
              case -200:
              case -300:
                that.failureResult()
                break
              case -101:
                that.failureResult()
                break
              default:
                that.failureResult()
            }
          })
          .catch(err => {
            document.getElementById('stepOne').className = 'stepOne header'
            setTimeout(() => {
              this.stepOne = false
            }, 1000)
            console.log(err)
            this.failureResult()
          })
      }, 0)
    },
    // 显示结果
    // 备份失败
    failureResult() {
      this.failureTitle = '备份失败'
      this.failureDetail = '请检查您的机床和手机的网络连接状态'
      document.title = '备份失败'
      this.failure = true
      // 
      setTimeout(() => {
        document.getElementById('failure').className = 'failure header toShow'
      }, 1000)
    },
    // 备份成功
    successResult() {
      this.successTitle = '备份完成'
      document.title = '备份完成'
      this.success = true
      setTimeout(() => {
        document.getElementById('success').className = 'success header toShow'
      }, 1000)
      setTimeout(() => {
        this.$router.push({
          name: 'BackupList',
          params: this.$route.query.gmId
        })
      }, 2500)
    },
